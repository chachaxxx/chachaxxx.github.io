什么是this

JavaScript代码运行时创建执行上下文时会进行this绑定、创建词法环境和变量环境。

this时当前执行上下文（global、function 或 eval）的一个属性，在非严格模式下，总是指向一个对象，在严格模式下可以是任意值。

  • 全局上下文：在全局环境中，指向window。可以用globalThis获取

  • 函数上下文：this的值取决于函数被调用的方式，一般情况下调用时决定。

  • 类上下文：构造函数中this包含非静态的方法。

  • 派生类上下文：派生类的构造函数没有初始的 this 绑定。

怎么判断this

  • 理解：this 永远指向最后调用它的那个对象，没有的话指向全局对象。箭头函数比较特殊。
  • 规则：new绑定 > 显式绑定 > 隐式绑定 > 默认绑定
    • 默认绑定（函数方法在调用时默认全局）
    • 隐式绑定（调用函数的为对象，则指向对象）
    • 硬绑定 （bind、apply、call强行指定）
    • new绑定 （在new对象的时候会把新对象绑定到this上）
  • 总结：
    1. 在对象方法中，this 指的是调用对象。
    2. 单独的情况下，this 指的是全局对象。
    3. 在函数中，this 指的是全局对象。
    4. 在函数中，严格模式下，this 是 undefined。
    5. 在DOM事件或DOM内联中，this 指的是接收事件的元素。
    6. 延时函数，定时调用函数this 指的是全局对象。
    7. 箭头函数this指的定义的时候的父上下文的所在上下文，它没有this，不能当做构造函数。
    8. 像 call() 和 apply() 这样的方法可以将 this 引用到任何对象。

bind、apply、call

  • 都可以强行指定函数的this
  • bind返回的是函数方法，apply,call返回的是函数调用结果。
  • apply使用数组参数，call使用列表参数

<script>
  /* 缓存器 */
  let man = {
    name: 'Tom',
    tel: {
      pre: '+086',
      num: '778',
    },
    phone(num) {
      console.log(
        `${this.name} with ${this.tel.pre}${this.tel.num} call ${num}`
      )
    },
  }

  function decorate(func) {
    return function (num) {
      let res = func.apply(this, [num]) // call、apply返回调用结果
      // let res = func.call(this, num) // call、apply返回调用结果
      // let res = func.bind(this) // 这个返回的是方法
      // let res = func(num) // this会丢失指向
      return res
    }
  }

  function decorateBind(func) {
    // 如果 bind 函数的参数列表为空，或者thisArg是null或undefined，执行作用域的 this 将被视为新函数的 thisArg
    // return func.bind(thisArg) // 这样执行了函数，但是无法定义参数
    return function (num) {
      // return func.bind(this, num) // 这样是返回function
      return func.bind(this, num)(num) //这样是运行返回的function(立即执行函数)
    }
  }

  // 缓存器
  function cachingDecorate(func) {
    let temp = 'test' // 没有返回的chrome不会存进去
    let cache = new Map() // 闭包，存在封装方法的[[Scoped]]中
    return function (x) {
      if (cache.has(x)) {
        return cache.get(x)
      }
      let result = func.call(this, x)
      cache.set(x, result)
      return result
    }
  }

  // 获取缓存的缓存器(结合间谍包装器)
  function cachingDecorateShow(func) {
    wrapper.cache = new Map() // 函数对象
    function wrapper(x) {
      //  wrapper.cache = new Map()
      if (wrapper.cache.has(x)) {
        return wrapper.cache.get(x)
      }
      let result = func.call(this, x)
      wrapper.cache.set(x, result)
      return result
    }
    return wrapper
  }

  man.phone1 = decorate(man.phone)
  man.phone2 = decorateBind(man.phone)
  man.phone3 = cachingDecorate(man.phone)
  man.phone4 = cachingDecorateShow(man.phone)
  man.phone1('666')
  man.phone2('666')
  man.phone3('666')
  man.phone4('666')
  man.phone4('777')
  console.log(man.phone4.cache)
</script>

防抖（debounce）

连续触发事件，一定时间内没有再次触发就会执行一次

时间内触发了就会重新延时

使用：搜索栏文字输入防抖，游戏回城事件

节流（throttle）

一段时间内只能触发一次，如果这段时间内触发多次事件，只有第一次生效会触发回调函数，一段时间过后才能再次触发（n秒内第一次生效）。

<script>
  /*  防抖装饰器的结果是一个包装器，该包装器将暂停对 f 的调用，
    直到经过 ms 毫秒的非活动状态（没有函数调用，“冷却期”），
    然后使用最新的参数调用 f 一次。 */
  // _.debounce(func, [(wait = 0)], [(options = {})]) //lodash
  // 手写
  function debounce(func, ms) {
    let timeout // 将timer保存在debounce实例的[[Scoped]]
    return function () {
      clearTimeout(timeout) // 每点一次清除一次定时器,说不定还没有进行func调用
      timeout = setTimeout(
          () => func.apply(this, arguments), // 到了定时器时间，才会执行一次
          ms) // 重启定时器
    }
  }

  /* 节流装饰器的结果是一个包装器
  该包装器将在f多次调用时
  会在每 ms 毫秒最多将调用传递给 f 一次 */
  // _.throttle(func, [(wait = 0)], [(options = {})])
  // 手写
  function throttle(func, ms) {
    let isThrottled = false,
      savedArgs,
      savedThis

    function wrapper() {
      if (isThrottled) {
        // 在这种状态下，所有调用都记忆在 savedArgs/savedThis 中。请注意，上下文和参数（arguments）同等重要，应该被记下来。我们同时需要他们以重现调用
        savedArgs = arguments // function参数
        savedThis = this
        return
      }
      // 调用包装func
      func.apply(this, arguments) // 在第一次调用期间，wrapper 只运行 func 并设置冷却状态（isThrottled = true）。
    
      // 未开始节流
      isThrottled = true
    
      setTimeout(function () {
        isThrottled = false // 然后经过 ms 毫秒后，触发 setTimeout。冷却状态被移除（isThrottled = false），如果我们忽略了调用，则将使用最后记忆的参数和上下文执行 wrapper。
        if (savedArgs) {
          wrapper.apply(savedThis, savedArgs) // 运行最后被throttle的调用，若无这一行,没有最后一次不满足冷却时间的输出 [1,11,19] || [1,11]
          savedArgs = savedThis = null
        }
      }, ms)
    }
    
    return wrapper
  }

  man.phone = throttle(man.phone, 1000)
  let num = 1
  let id = this.setInterval(() => man.phone(num++), 100)
  this.setTimeout(() => {
    clearInterval(id)
  }, 1900)
</script>
