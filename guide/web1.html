树形表格勾选

响应快，功能全，不卡顿的树形表格勾选功能。

     /**
     * @description: 表格配置变化接收
     * @param {*} record 变更数据项
     * @param {*} key 配置项
     * @param {*} type 变更方向（横向，竖向）
     * @return {*}
     */
    changeCheck(record, key, type) {
      // 更新动画加载
      this.isSetting = true
      if (type == 'row') {
        // 横向
        let state = Number(!record[key])
        // 取消时,判断子节点，如果有设置就拦截
        if (!state && record.Children && record.Children.length && this.isChildSet(record.Children, key)) {
          this.$notification.warning({ message: '子节点已设置该计划配置', class: 'pos-center' })
          this.isSetting = false
          return
        }
        // 获取数据源中相关节点(过滤相关第一层节点，精简数据)
        let list = this.$refs.dataTable.dataList.filter((x) => record.ParentCode.indexOf(x.RuleCode) > -1)
        // ***** 重点 递归设置所有父子相关节点*****
        this.setSelfAndParentsCheck(list, record, key, state)
      } else {
        // 竖向设置所有数据的某一字段
        record.checked = !record.checked
        this.setConfigAllCheck(key, record.checked)
      }
      this.isSetting = false
    },

下拉选择表格

antd 组件中没有实现表格式的下拉选择器。其在数据较多、显示多列数据的表单中使用很广泛。

利用 antd 现有的组件进行组合，对样式等进行改造，实现了通用的下拉选择表格。

<template>
  <a-dropdown-button
    class="customer-dropdown"
    slot="addonAfter"
    placement="bottomLeft"
    :visible="menuVisible"
    @click="buttonClick"
    :getPopupContainer="
      triggerNode => {
        return triggerNode.parentNode
      }
    "
  >
    <a-input
      v-model="cnName"
      placeholder="请选择"
      @change="InputChange"
      style="width:100%"
    ></a-input>
    <a-menu slot="overlay">
      <a-menu-item class="customer-menu">
        <a-tree-select
          :disabled="isorg"
          placeholder=""
          v-model="orgCode"
          :tree-data="treeData"
          style="width:100%"
        />
        <a-input-search
          @search="onSearch"
          v-model="keyword"
          placeholder="用户名"
          size="small"
        />
        <a-table
          class="drop-table"
          ref="table"
          size="small"
          :scroll="{ x: true }"
          :rowKey="record => record.Id"
          :dataSource="loadData"
          :columns="columns"
          :customRow="onClickRow"
          :pagination="pagination"
          @change="onPageChange"
        >
          <a slot="name" slot-scope="text">{{ text }}</a>
        </a-table>
      </a-menu-item>
    </a-menu>
    <a-icon slot="icon" type="user" />
  </a-dropdown-button>
</template>
