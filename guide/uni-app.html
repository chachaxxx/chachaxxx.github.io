    if (
      config.url.substr(-authConfig().token_endpoint.length) !==
      authConfig().token_endpoint
    ) {
      const token = getToken()
      if (isLogin()) {
        // 判断是否存在token，如果存在的话，则每个http header都加上token
        config.header['Authorization'] = 'Bearer ' + token
      } else if (!config.url.startsWith('http')) {

  // 请求错误
  (config) => {
    return Promise.reject(config)
  }
)

路由的配置

uni-app的路由用法与 Vue Router 不同，为框架统一管理，在pages.json里配置每个路由页面的路径及页面样式。

对于页面权限管理，uni-app不使用动态路由。而是和安卓类似，首先在page.json注册每一个页面，然后通过获取并存储网络授权菜单，通过vuex展示相应的页面入口。

vant可以使用vue Router。使用本地路由+授权动态路由进行配置。

/**
 * 获取路由菜单 -- 嵌在路由守卫里面
 * @param {*} to
 * @param {*} from
 * @param {*} next
 */
const loadMenus = (to, from, next) => {
  store
    .dispatch('GenerateRoutes', {
      roleId: store.getters.currentRole.Id || storage.get(CURRENT_ROLE).Id,
      userId: store.getters.userInfo.Id || storage.get(USER_INFO).Id
    })
    .then(() => {
      // 根据roles权限生成可访问的路由表
      // 动态添加可访问路由表
      router.addRoutes(store.getters.addRouters) // 存在内存的router
      // 请求带有 redirect 重定向时，登录自动重定向到该地址
      // storage.set(APP_REDIRECT, to.path)
      const redirect = decodeURIComponent(storage.get(APP_REDIRECT) || to.path)
      if (to.path === redirect) {
        // set the replace: true so the navigation will not leave a history record
        next({ ...to, replace: true })
      } else {
        // 跳转到目的路由
        next({ path: redirect, replace: true })
      }
    })
}

常用数据的vuex配置

对经常使用、全局性的数据进行状态管理，保持数据的全局一致。稍微简单的数据结构可以使用store进行简单管理。

// 获取用户授权菜单
const permission = {
  state: {
    routers: constantRouterMap // 本地菜单,
    addRouters: [],
    isAddRouters: false
  },
  mutations: {
    SET_ROUTERS: (state, routers) => {
      state.addRouters = routers
      state.routers = constantRouterMap.concat(routers)
      storage.set(ADD_ROUTERS, routers)
      state.isAddRouters = true
    }
  },
  actions: {
    GenerateRoutes({ commit }, data) {

        // 更多操作 ......
        commit('SET_ROUTERS', accessedRouters)
        resolve()
      })

  }
}

/**
 * 判断文件格式
 * @param {String} filename
 * @returns {string}
 */
Vue.filter('testExt', function(filename) {
	const name = filename

	let dit = name.lastIndexOf(".");
	let suffix = name.substring(dit + 1, name.length);

	let isExt = ['jpg', 'jpeg', 'png', 'bmp', 'doc', 'docx', 'pdf', 'txt'].indexOf(suffix) >= 0;

	return isExt
})

/**
 * 设置上一界面展开时刷新---加入全局
 */
Vue.prototype.$onSetLastRefresh = onSetLastRefresh
export function onSetLastRefresh() {
	try {
		let pages = getCurrentPages();
		let currPage = pages[pages.length - 1]; //当前页面
		let prevPage = pages[pages.length - 2]; //上一个页面

        // uni-APP条件编译
		// #ifdef APP-PLUS
		prevPage.$vm._isRefresh = true;
		// #endif

		// #ifdef H5
		prevPage._data._isRefresh = true;
		// #endif
	} catch (e) {
		console.info('编译成功刷新界面后，丢失上级界面，此时上级界面刷新不可用');
	}
}
